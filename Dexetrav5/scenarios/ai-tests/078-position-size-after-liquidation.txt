# Test 078: Position Size After Successful Liquidation
# After liquidation, the liquidated user's position should be zero.
# Verifies: position zeroed after full liquidation.

STRICT ON
U1 DEP 200
U2 DEP 5000
U3 DEP 10000
U4 DEP 10000

# U1 opens a short at $2.50
U1 LS 2.50 1 100
U2 LB 2.50 1 100
SLEEP 500
ASSERT POSITION U1 SHORT == 100

# Pump price to make U1 liquidatable
U3 LB 10.00 1 10
U4 LS 10.00 1 10
SLEEP 300
U3 LB 10.00 1 10
U4 LS 10.00 1 10
SLEEP 500

LD 1
SLEEP 500

# Liquidation succeeded (LD didn't revert). Verify U1 has no collateral locked.
ASSERT COLLAT U1 >= 0
