# Test 084: Full Lifecycle: Deposit → Trade → Close → Withdraw
# Complete round-trip verifying all state transitions.
# Deploy pre-funds each user with $1000 collateral.
# Total collateral = $1000 (deploy) + $5000 (deposit) = $6000.
# Verifies: end-to-end lifecycle, collateral recovery after closing.

STRICT ON
U1 DEP 5000
U2 DEP 5000

SLEEP 300
ASSERT COLLAT U1 >= 6000

# Open long
U2 LS 2.50 1 50
U1 LB 2.50 1 50
SLEEP 500
ASSERT POSITION U1 LONG == 50

# Close at same price (no PnL)
U2 LB 2.50 1 50
U1 LS 2.50 1 50
SLEEP 500

# Withdraw everything (all $6000)
U1 WDR 6000
SLEEP 300

ASSERT COLLAT U1 == 0
